// index.js — Sistem Modülü İstemci Katmanı
// Düzeltmeler: tekrarlayan listener'lar birleştirildi,
// cancelAnimationFrame kaldırıldı, author null koruması eklendi

"use strict";

const system = require("./system.json");

const name                   = system.name;
const version                = system.version;
const author                 = system.author ?? "Bilinmiyor";
const AWSTemplateFormatVersion = system.AWSTemplateFormatVersion ?? null;

/** Standart çıktı nesnesi */
const output = {
  system: { name, version, author, AWSTemplateFormatVersion }
};

// Tüm DOM işlemleri tek listener'da toplandı
document.addEventListener("DOMContentLoaded", () => {
  function setById(id, value) {
    const el = document.getElementById(id);
    if (el) el.textContent = value;
  }

  const outputEl = document.getElementById("output");
  if (outputEl) outputEl.textContent = JSON.stringify(output, null, 2);

  setById("name",    name);
  setById("version", version);
  setById("author",  author);
});

module.exports = { system, output, name, version, author };

/// <summary>
/// SysController — Sistem bilgilerini sunan ASP.NET Core denetleyicisi.
/// Rota: /sys  |  Endpoint'ler: Index, Durum, Versiyon, Error
/// </summary>
[ApiController]
[Route("[controller]")]
public class SysController : Controller
{
  private readonly ILogger<SysController> _logger;

  public SysController(ILogger<SysController> logger)
    => _logger = logger ?? throw new ArgumentNullException(nameof(logger));

  /// GET /sys — Ana görünüm
  [HttpGet]
  public IActionResult Index() => View();

  /// GET /sys/durum — Sistem durumu JSON
  [HttpGet("durum")]
  public IActionResult Durum() => Ok(new {
    durum  = "çalışıyor",
    zaman  = DateTime.UtcNow.ToString("o"),
    makine = Environment.MachineName
  });

  /// GET /sys/versiyon — Versiyon bilgisi
  [HttpGet("versiyon")]
  public IActionResult Versiyon() => Ok(new {
    uygulama = "1.0.0",
    dotnet   = Environment.Version.ToString(),
    os       = Environment.OSVersion.VersionString
  });

  /// GET /sys/hata — Hata sayfası
  [HttpGet("hata")]
  [ResponseCache(Duration = 0, NoStore = true)]
  public IActionResult Error() => View("Error");
}

/**
 * AppSystem — Sistem kodu üretici sınıf.
 *
 * Örnek çıktı: SYS-20251029143022-a3f9c1d8
 *
 * ⚠ Orijinal class adı 'system' java.lang.System ile çakışıyordu.
 *    Yeni ad: AppSystem
 *
 * @version 1.0.0
 */
public class AppSystem {

  private static final String ONEK = "SYS";
  private static final DateTimeFormatter FMT =
    DateTimeFormatter.ofPattern("yyyyMMddHHmmss");

  /** Giriş noktası */
  public static void main(String[] args) {
    System.out.println("Üretilen: " + generateSystemCode());
  }

  /**
   * Tekil sistem kodu üretir.
   * Biçim: SYS-{yyyyMMddHHmmss}-{uuid[0..8]}
   *
   * @return Tekil sistem kodu
   */
  public static String generateSystemCode() {
    String zaman    = LocalDateTime.now().format(FMT);
    String uuidPart = UUID.randomUUID().toString().substring(0, 8);
    return ONEK + "-" + zaman + "-" + uuidPart;
  }

  /**
   * Kod biçimini doğrular: SYS-14rakam-8hexkarakter
   *
   * @param  kod Doğrulanacak string
   * @return true → geçerli
   */
  public static boolean gecerliMi(String kod) {
    if (kod == null) throw new IllegalArgumentException("null olamaz");
    return kod.matches("^SYS-\\d{14}-[a-f0-9]{8}$");
  }
}

"""
system.py — Sistem Yardımcıları
Düzeltmeler:
  - s2clientprotocol / sc2reader import'ları kaldırıldı
  - Program.Progressbar(): staticmethod sözdizimi düzeltildi
  - assert_line_data @staticmethod decorator eklendi
"""
import os, sys, platform
from datetime import datetime


class System:
  """Sistem bilgisi ve yardımcı metotları içeren sınıf."""

  def main(self) -> None:
    """Sistem bilgilerini ekrana basar."""
    dir_path = os.path.dirname(os.path.realpath(__file__))
    print(f"Dizin  : {dir_path}")
    print(f"Python : {sys.version.split()[0]}")
    print(f"OS     : {platform.system()} {platform.release()}")

  @staticmethod
  def assert_line_data(flag: int = 1) -> None:
    """
    Bayrak değerini doğrular.
    :param flag: Kontrol edilecek değer (varsayılan: 1)
    :raises AssertionError: flag 1 değilse
    """
    assert flag == 1, "Flag 1 olmalıdır!"


class Program:
  """Uygulama başlatıcı."""

  @staticmethod           # ← orijinalde sözdizimi hatasıydı
  def progressbar(toplam: int = 10) -> None:
    """Metin tabanlı ilerleme çubuğu."""
    import time
    for i in range(toplam + 1):
      bar = "█" * i + "░" * (toplam - i)
      print(f"\r[{bar}] {i}/{toplam}", end="", flush=True)
      time.sleep(0.05)
    print()

  @staticmethod
  def main() -> None:
    System().main()
    System.assert_line_data(1)


if __name__ == "__main__":
  Program.main()
